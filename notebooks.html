    .setInputCols(["spell", "sentence"]) \
    .setOutputCol("sentiment") \

    .setNegativeSource("../negative") \

                                                                                                         In [ ]:
pipeline = Pipeline(stages=[
    document_assembler,
    tokenizer,
    spell_checker,
    sentence_detector,
    sentiment_detector
])

def _extract_metadata(rows):
    r = []
    for row in rows:
        row_dict = row.asDict()
        r.append(list(row_dict["metadata"].values()))
    return r
extract_metadata = udf(_extract_metadata, returnType=ArrayType(ArrayType(StringType())))

sentiment_data = pipeline.fit(data).transform(data).withColumn("sent", extract_metadata("sentiment")).select("sent")
